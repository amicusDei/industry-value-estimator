```python
def prepare_prophet_from_anchors(segment: str) -> pd.DataFrame:
    """Prepare market anchors data in Prophet ds/y format for a segment.
    Filters to real observations (n_sources > 0). Returns DataFrame with ds, y columns."""
    from config.settings import DATA_PROCESSED
    anchors = pd.read_parquet(DATA_PROCESSED / "market_anchors_ai.parquet")
    real = anchors[anchors["n_sources"] > 0].copy()
    seg = (
        real[real["segment"] == segment]
        .sort_values("estimate_year")
        [["estimate_year", "median_real_2020"]]
        .rename(columns={"estimate_year": "ds", "median_real_2020": "y"})
    )
    seg["ds"] = pd.to_datetime(seg["ds"].astype(str) + "-01-01")
    return seg
```

```python
def fit_prophet_from_anchors(segment: str, changepoint_year: int = 2022) -> Prophet:
```

```python
def compute_source_disagreement_columns(
    forecast_df: pd.DataFrame,
    anchors_df: pd.DataFrame,
) -> pd.DataFrame:
    """Add source disagreement band columns from market anchors.
    Only populated for historical years where anchor data exists."""
    anchor_lookup = anchors_df.set_index(["estimate_year", "segment"])[["p25_real_2020", "p75_real_2020"]]
    df = forecast_df.copy()
    df["anchor_p25_real_2020"] = df.apply(
        lambda r: anchor_lookup.loc[(r["year"], r["segment"]), "p25_real_2020"]
        if (r["year"], r["segment"]) in anchor_lookup.index else float("nan"),
        axis=1,
    )
    df["anchor_p75_real_2020"] = df.apply(
        lambda r: anchor_lookup.loc[(r["year"], r["segment"]), "p75_real_2020"]
        if (r["year"], r["segment"]) in anchor_lookup.index else float("nan"),
        axis=1,
    )
    return df
```
