2. Add a `verify_cagr_range` function to `forecast.py`:

```python
def verify_cagr_range(
    df: pd.DataFrame,
    segments: list[str],
    start_year: int = 2025,
    end_year: int = 2030,
    min_cagr: float = 0.15,
    max_cagr: float = 0.60,
) -> dict[str, float]:
    """Verify per-segment CAGR is in expected range.
    Returns dict of {segment: cagr_pct}.
    Logs warning if any segment is outside min_cagr-max_cagr range.
    """
    import logging
    logger = logging.getLogger(__name__)
    results = {}
    for seg in segments:
        seg_df = df[df["segment"] == seg].sort_values("year")
        val_start = seg_df[seg_df["year"] == start_year]["point_estimate_real_2020"]
        val_end = seg_df[seg_df["year"] == end_year]["point_estimate_real_2020"]
        if len(val_start) > 0 and len(val_end) > 0:
            cagr = (val_end.iloc[0] / val_start.iloc[0]) ** (1 / (end_year - start_year)) - 1
            results[seg] = cagr
            if not (min_cagr <= cagr <= max_cagr):
                logger.warning(
                    f"Segment {seg} CAGR {cagr:.1%} outside target range "
                    f"[{min_cagr:.0%}, {max_cagr:.0%}]"
                )
    return results
```
