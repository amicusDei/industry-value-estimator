**Part B: Load real forecast data for key findings**

Before writing the paper or README, load the real forecast data to extract headline numbers:
```python
uv run python -c "
import sys; sys.path.insert(0, '.')
import pandas as pd
import yaml
from config.settings import DATA_PROCESSED
df = pd.read_parquet(DATA_PROCESSED / 'forecasts_ensemble.parquet')
with open('config/industries/ai.yaml') as f:
    cfg = yaml.safe_load(f)
vc = cfg['value_chain']
anchor_year = int(vc['anchor_year'])
anchor_total = float(vc['anchor_value_usd_billions'])
shares = vc['segment_anchor_shares']
# Compute multipliers
anchor_rows = df[df['year'] == anchor_year]
global_idx = anchor_rows['point_estimate_real_2020'].sum()
global_mult = anchor_total / global_idx if global_idx != 0 else 1.0
# 2024 and 2030 values
for year in [2024, 2030]:
    yr_rows = df[df['year'] == year]
    total_idx = yr_rows['point_estimate_real_2020'].sum()
    total_usd = total_idx * global_mult
    print(f'Year {year}: \${total_usd:.0f}B (index: {total_idx:.2f})')
    for seg in sorted(df['segment'].unique()):
        seg_rows = yr_rows[yr_rows['segment'] == seg]
        seg_idx = float(seg_rows['point_estimate_real_2020'].iloc[0])
        seg_usd = seg_idx * global_mult * shares.get(seg, 0.25)
        print(f'  {seg}: \${seg_usd:.0f}B')
"
```
